# Field with type

SomeModel
  field: int

==>

root(Model(ModelName, Field(FieldName, Type(TypeSymbol))))

# Field with enum type

SomeModel
  field: BLUE, GREEN

==>

root(Model(ModelName, Field(FieldName, Type(TypeEnumVariant,TypeEnumVariant))))

# Field with union type

SomeModel
  field: int | BLUE, GREEN

==>

root(Model(ModelName, Field(FieldName, Type(UnionType(TypeSymbol, TypeEnumVariant, TypeEnumVariant)))))

# Field with complex type

SomeModel
  field: BLUE, GREEN | int | [string | date ]

==>

root(Model(ModelName, Field(FieldName,
  Type(UnionType(TypeEnumVariant, TypeEnumVariant, TypeSymbol, ArrayOfType(OpenBracket, UnionType(TypeSymbol,TypeSymbol), CloseBracket)))
)))
